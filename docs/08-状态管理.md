# 08 - 状态管理

## 概述

Claude Code 使用**自定义的轻量级 Store**（类似 Zustand）管理全局状态，结合 React Context 和 `useSyncExternalStore` 实现组件间的响应式数据流。

- **Store 实现**：`state/store.ts`
- **全局状态**：`state/AppStateStore.ts`
- **Provider 和 Hook**：`state/AppState.tsx`
- **React Context**：`context/`（9+ 个上下文）
- **自定义 Hook**：`hooks/`（80+ 个 Hook）

## 自定义 Store

### 核心实现

```typescript
// state/store.ts
type Listener = () => void
type OnChange<T> = (args: { newState: T; oldState: T }) => void

type Store<T> = {
  getState: () => T
  setState: (updater: (prev: T) => T) => void
  subscribe: (listener: Listener) => () => void
}

function createStore<T>(initialState: T, onChange?: OnChange<T>): Store<T> {
  let state = initialState
  const listeners = new Set<Listener>()

  return {
    getState: () => state,

    setState: (updater) => {
      const prev = state
      const next = updater(prev)
      if (Object.is(next, prev)) return   // 跳过无变化更新
      state = next
      onChange?.({ newState: next, oldState: prev })
      for (const listener of listeners) listener()
    },

    subscribe: (listener) => {
      listeners.add(listener)
      return () => listeners.delete(listener)  // 返回取消订阅函数
    },
  }
}
```

### 设计特点

| 特性 | 说明 |
|------|------|
| **不可变更新** | 使用函数式 updater，不直接修改 state |
| **Object.is 优化** | 相同引用跳过更新和通知 |
| **onChange 回调** | 全局变更钩子，用于副作用 |
| **Set 监听器** | 自动去重，O(1) 添加/移除 |
| **泛型** | 完全类型安全 |

## Provider & Hook

### AppStateProvider

```typescript
// state/AppState.tsx
const AppStoreContext = React.createContext<AppStateStore | null>(null)

function AppStateProvider({ children, initialState, onChangeAppState }) {
  const [store] = useState(
    () => createStore(initialState ?? getDefaultAppState(), onChangeAppState)
  )
  return (
    <HasAppStateContext.Provider value={true}>
      <AppStoreContext.Provider value={store}>
        <MailboxProvider>
          <VoiceProvider>
            {children}
          </VoiceProvider>
        </MailboxProvider>
      </AppStoreContext.Provider>
    </HasAppStateContext.Provider>
  )
}
```

### Selector Hook

```typescript
// 使用 useSyncExternalStore 实现无撕裂读取
function useAppState<T>(selector: (s: AppState) => T): T {
  const store = useAppStore()
  const get = () => selector(store.getState())
  return useSyncExternalStore(store.subscribe, get, get)
}

// 使用示例
const model = useAppState(s => s.mainLoopModel)
const mode = useAppState(s => s.toolPermissionContext.mode)
const tasks = useAppState(s => s.tasks)
```

## onChange 副作用

### 状态变更侧效应

```typescript
// state/onChangeAppState.ts
function onChangeAppState({ newState, oldState }) {
  // 权限模式变更 → 通知远程会话
  if (newState.toolPermissionContext.mode !== oldState.toolPermissionContext.mode) {
    notifySessionMetadataChanged()
    notifyPermissionModeChanged()
  }

  // 模型变更 → 持久化到用户设置
  if (newState.mainLoopModel !== oldState.mainLoopModel) {
    persistToUserSettings(newState.mainLoopModel)
  }

  // 展开视图变更 → 持久化到全局配置
  if (newState.expandedView !== oldState.expandedView) {
    persistToGlobalConfig('showExpandedTodos', newState.expandedView)
  }

  // verbose 变更 → 持久化
  if (newState.verbose !== oldState.verbose) {
    persistToGlobalConfig('verbose', newState.verbose)
  }
}
```
